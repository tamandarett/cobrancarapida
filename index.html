function atualizarContadores(agrupados) { const total = Object.keys(agrupados).length; let ca = 0; let ft = 0; let semFone = 0; Object.values(agrupados).forEach(g => { const foneData = processarTelefone(g.fone); if (!foneData.valido) semFone++; // CORREÇÃO 1: Lógica robusta para identificar o grupo do cliente const temTitulosFT = g.titulos.some(t => t.num.toUpperCase().includes("D") || ['FT', 'DP', 'IX', 'CP', 'RN'].includes(t.fpOriginal) ); const temTitulosCA = g.titulos.some(t => t.fpOriginal === 'CA'); if (temTitulosFT) { ft++; } else if (temTitulosCA) { ca++; } }); document.getElementById('statTotal').innerText = total; document.getElementById('statCA').innerText = ca; document.getElementById('statFT').innerText = ft; document.getElementById('statSemFone').innerText = semFone; } function exportarSemFone() { // CORREÇÃO 2: Aplicando a busca (term) na exportação const term = document.getElementById('search').value.toLowerCase(); let filtrados = rawData.filter(item => { const dias = parseInt(item["DIAS EMATRASO"]) || 0; const fp = normalizarFP(item); const foneData = processarTelefone(item["FONE"]); // Verifica se o fone é inválido/fixo const foneInvalido = !foneData.valido; // Filtro por modo (CA ou FT) let passaFiltroModo = false; if (currentMode === 'CA') { passaFiltroModo = (fp === 'CA'); } else { passaFiltroModo = (['FT', 'DP', 'IX', 'CP', 'RN'].includes(fp) && dias >= 4); } // Filtro por termo de busca const passaBusca = !term || item["NOME DO CLIENTE"]?.toLowerCase().includes(term) || item["CÓDIGO"]?.toString().includes(term); return passaFiltroModo && foneInvalido && passaBusca; }); const semFoneList = []; const seen = new Set(); filtrados.forEach(item => { if (!seen.has(item["CPF/CNPJ"])) { semFoneList.push(`Código: ${item["CÓDIGO"]} | Cliente: ${item["NOME DO CLIENTE"].trim()} | Fone no Cadastro: ${item["FONE"] || "Vazio"}`); seen.add(item["CPF/CNPJ"]); } }); if (semFoneList.length === 0) return mostrarErro("Nenhum cliente sem WhatsApp encontrado nos filtros/busca atuais."); const blob = new Blob([semFoneList.join('\n')], { type: 'text/plain' }); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = `clientes_sem_whatsapp.txt`; a.click(); }
